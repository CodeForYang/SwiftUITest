import XCTest
@testable import SwiftUITest

final class SwiftUITestTests: XCTestCase {

    override func setUpWithError() throws {
        continueAfterFailure = false
    }

    // MARK: - JSON Decoding

    func testLandmarkDecoding() throws {
        // Given
        let url = try XCTUnwrap(
            Bundle.main.url(forResource: "landmarkData", withExtension: "json")
        )
        let data = try Data(contentsOf: url)

        // When
        let landmarks = try JSONDecoder().decode([Landmark].self, from: data)

        // Then
        XCTAssertEqual(landmarks.count, 12)
        let first = landmarks[0]
        XCTAssertEqual(first.name, "Turtle Rock")
        XCTAssertEqual(first.park, "Joshua Tree National Park")
        XCTAssertEqual(first.category, .rivers)
        XCTAssertTrue(first.isFeatured)
        XCTAssertTrue(first.isFavorite)
    }

    // MARK: - ModelData Computed Properties

    func testModelDataFeatures() throws {
        // Given
        let modelData = ModelData()

        // When
        let features = modelData.features

        // Then
        XCTAssertFalse(features.isEmpty)
        for landmark in features {
            XCTAssertTrue(landmark.isFeatured)
        }
        XCTAssertLessThanOrEqual(features.count, modelData.landmarks.count)
    }

    func testModelDataCategories() throws {
        // Given
        let modelData = ModelData()

        // When
        let categories = modelData.categories

        // Then
        XCTAssertEqual(Set(categories.keys), Set(["Lakes", "Rivers", "Mountains"]))
        for (key, landmarks) in categories {
            for landmark in landmarks {
                XCTAssertEqual(landmark.category.rawValue, key)
            }
        }
    }
}
```

import XCTest

final class SwiftUITestUITests: XCTestCase {

    var app: XCUIApplication!

    override func setUpWithError() throws {
        continueAfterFailure = false
        app = XCUIApplication()
        app.launch()
    }

    // MARK: - Tab Switching

    func testTabSwitching() throws {
        // Switch to List tab
        let listTab = app.buttons["list"]
        XCTAssertTrue(listTab.waitForExistence(timeout: 3))
        listTab.tap()

        // Verify list content appears
        let landmarksTitle = app.staticTexts["Landmarks"]
        XCTAssertTrue(landmarksTitle.waitForExistence(timeout: 3))

        // Switch back to Featured tab
        let featuredTab = app.buttons["Featured"]
        XCTAssertTrue(featuredTab.exists)
        featuredTab.tap()

        // Verify featured content appears
        let featuredText = app.staticTexts["Featured"]
        XCTAssertTrue(featuredText.waitForExistence(timeout: 3))
    }

    // MARK: - Landmark Navigation

    func testLandmarkNavigation() throws {
        // Navigate to List tab
        app.buttons["list"].tap()
        XCTAssertTrue(app.staticTexts["Landmarks"].waitForExistence(timeout: 3))

        // Tap first landmark cell
        let firstLandmark = app.buttons["Turtle Rock"]
        XCTAssertTrue(firstLandmark.waitForExistence(timeout: 3))
        firstLandmark.tap()

        // Verify detail view
        let detailTitle = app.staticTexts["Turtle Rock"]
        XCTAssertTrue(detailTitle.exists)

        // Tap back
        let backButton = app.buttons["Landmarks"]
        XCTAssertTrue(backButton.exists)
        backButton.tap()

        // Verify back on list
        XCTAssertTrue(app.staticTexts["Landmarks"].waitForExistence(timeout: 3))
    }

    // MARK: - Favorite Filter

    func testFavoriteFilter() throws {
        // Navigate to List tab
        app.buttons["list"].tap()
        XCTAssertTrue(app.staticTexts["Landmarks"].waitForExistence(timeout: 3))

        let favoritesToggle = app.switches["Favorites only"]
        XCTAssertTrue(favoritesToggle.waitForExistence(timeout: 3))

        // Turn toggle on
        if favoritesToggle.value as? String == "0" {
            favoritesToggle.tap()
        }

        // Verify favorited items are visible (star.fill icon present)
        XCTAssertTrue(app.images["star.fill"].exists)

        // Turn toggle off
        if favoritesToggle.value as? String == "1" {
            favoritesToggle.tap()
        }

        // Verify all landmarks visible again
        XCTAssertTrue(app.buttons["Turtle Rock"].exists)
    }
}
```
